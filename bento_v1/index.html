Google Apps Script (GAS) 部署教學與代碼

本訂餐系統採用完全前後端分離設計。前端透過 google.script.run 呼叫 GAS 後端，並將資料保存在 Google Sheet（試算表）中。以下為部署至真實 GAS 雲端的步驟：

1

建立 Google 試算表

在您的 Google 雲端硬碟建立一個新的「試算表」，並在其中手動建立兩個工作表分頁，分別命名為 Products 與 Orders（或者執行下面的 GAS 初始化代碼，系統會自動建立）。

2

開啟 Google Apps Script

在您的試算表選單中點選 擴充功能 ➡️ Apps Script，開啟指令碼編輯器。

3

複製並貼上後端程式碼 (`Code.gs`)

將預設的 myFunction 清空，複製下方編輯器中的 Code.gs 內容並完整貼上，按下儲存鍵。

4

建立前端網頁檔案 (`index.html`)

在 Apps Script 專案左側點選「+」新增一個 HTML 檔案，命名為 index（系統會自動生成 index.html），並將本專案的 index.html、style.css 與 script.js 合併（CSS 放入 <style>，JS 放入 <script>）後貼上。

5

發布 Web 應用程式

點選右上角 部署 ➡️ 新增部署。選取「網頁應用程式」，並設定：
- 專案執行為：我 (您的 Google 帳戶)
- 誰有權存取：任何人 (Anyone)
部署完成後，您即可獲得一個專屬的 Web App URL，直接線上點餐！

// 3. 取得所有商品
function getProducts() {
  initDatabase();
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Products");
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var products = [];
  
  for (var i = 1; i < data.length; i++) {
    var product = {};
    for (var j = 0; j < headers.length; j++) {
      product[headers[j]] = data[i][j];
    }
    products.push(product);
  }
  return products;
}

// 4. 提交新訂單
function submitOrder(orderData) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Orders");
  
  // 生成隨機訂單編號
  var orderId = "SLK-" + new Date().getTime().toString().slice(-6) + Math.floor(10 + Math.random() * 90);
  var timestamp = new Date();
  
  sheet.appendRow([
    orderId,
    timestamp,
    orderData.userEmail,
    orderData.customerName,
    orderData.phone,
    orderData.method,
    orderData.details,
    orderData.totalPrice,
    "pending", // 預設待處理
    orderData.remarks || ""
  ]);
  
  return { success: true, orderId: orderId };
}

// 6. 更新訂單狀態 (後台用)
function updateOrderStatus(orderId, newStatus) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Orders");
  var data = sheet.getDataRange().getValues();
  
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === orderId) {
      sheet.getRange(i + 1, 9).setValue(newStatus); // 狀態在第 9 欄
      return { success: true };
    }
  }
  return { success: false, message: "找不到該訂單" };
}

// 7. 更新商品資訊 (後台編輯用)
function updateProduct(productData) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Products");
  var data = sheet.getDataRange().getValues();
  
  for (var i = 1; i < data.length; i++) {
    if (data[i][0].toString() === productData.id.toString()) {
      sheet.getRange(i + 1, 2).setValue(productData.name);
      sheet.getRange(i + 1, 3).setValue(productData.category);
      sheet.getRange(i + 1, 4).setValue(productData.description);
      sheet.getRange(i + 1, 5).setValue(Number(productData.price));
      sheet.getRange(i + 1, 6).setValue(productData.image);
      sheet.getRange(i + 1, 7).setValue(productData.status);
      return { success: true };
    }
  }
  
  // 若找不到則新增
  sheet.appendRow([
    productData.id,
    productData.name,
    productData.category,
    productData.description,
    Number(productData.price),
    productData.image || "https://images.unsplash.com/photo-1546069901-ba9599a7e63c?w=300",
    productData.status || "instock"
  ]);
  return { success: true, isNew: true };
}
